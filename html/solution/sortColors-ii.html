class Solution {
    /**
     * @param colors: A list of integer
     * @param k: An integer
     * @return: nothing
     */
    public void sortColors2(int[] colors, int k) {
        // write your code here
        if ( colors == null || colors.length == 0 || k <=0 ) return;
        
        int l = 0;
        int r = colors.length - 1;
        int count = k;
        
        while( count > 0){
            
            // get local min and max
            int min = Integer.MAX_VALUE;
            int max = Integer.MIN_VALUE;
            for( int i=l; i<=r; i++){
                if ( colors[i] < min){
                    min = colors[i];
                }
                else if ( colors[i] > max){
                    max = colors[i];
                }
            }
            
            int cur = l;
            while( cur <= r){
                if ( colors[cur] == min ){
                    swap(colors, l, cur);
                    l++;
                    cur++;
                }
                else if ( colors[cur] > min && colors[cur] < max){
                    cur++;
                }else{
                    swap(colors, r, cur);
                    r--;
                }
            }
            
            count = count - 2;
        }
    }
    
    void swap(int[] arr, int i, int j){
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
