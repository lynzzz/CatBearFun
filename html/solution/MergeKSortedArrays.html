public class Solution {
    /**
     * @param arrays k sorted integer arrays
     * @return a sorted array
     */
    public List mergekSortedArrays(int[][] arrays) {
        // Write your code here
        if ( arrays == null || arrays.length == 0 ) return null;
        
        return helper( arrays, 0, arrays.length-1);
    }
    
    List helper(int[][] arrays, int start, int end){
        
        if ( start == end){
            ArrayList ret = new ArrayList();
            for( int i : arrays[start]){
                ret.add(i);
            }
            return ret;
        }
        
        int mid = start + ( end - start )/2;
        
        List left = helper(arrays, start, mid);
        List right = helper(arrays, mid + 1 , end);
        
        return mergeTwoSortedArray(left, right);
    }
    
    List mergeTwoSortedArray(List l, List r){
        
        int m = l.size();
        int n = r.size();
        int i=0;
        int j=0;

        List ret = new ArrayList();
        
        while( i<m && j<n){
            if ( l.get(i) < r.get(j) ){
                ret.add(l.get(i));
                i++;
            }else{
                ret.add(r.get(j));
                j++;
            }
        }
        
        while( i<m){
            ret.add(l.get(i));
            i++;
        }
        
        while( j<n){
            ret.add(r.get(j));
            j++;
        }
        
        return ret;
    }
}
Time O(kn) , where n is the average length of each array. 相当于全部数字过一遍 Space ??
