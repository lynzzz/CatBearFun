class Solution {

      boolean ret = false;

      public boolean wordBreak(String s, List<String> wordDict) {
          dfs(s, wordDict, 0);
          return ret;
      }

      void dfs(String s, List<String> wordDict, int pos){

          if ( pos == s.length()){
              ret = true;
              return;
          }


          for( int i=pos; i<s.length(); i++){
              String str = s.substring(pos, i + 1);
              if ( !wordDict.contains(str)){
                  continue;
              }

              dfs(s, wordDict, i+1);
          }

      }
  }
