class Solution {
    public int lengthOfLongestSubstringKDistinct(String s, int k) {

      if ( s == null || s.length() == 0 || k == 0) return 0;

              HashMap<Character, Integer> m = new HashMap<>();
              int start = 0;
              int ret = 0;
              int i=0;

              for( ; i<s.length(); i++){
                  char c = s.charAt(i);

                  if (m.size()<k){
                     m.put(c, m.getOrDefault(c, 0) + 1);
                  }else{
                      if ( m.containsKey(c)){
                          m.put(c, m.get(c) + 1);
                      }else{
                          ret = Math.max(ret, i-start);

                          char saveStart = s.charAt(start);

                          while( s.charAt(start) == saveStart){
                              m.put(saveStart, m.get(saveStart)-1);
                              start++;
                          }

                          if ( m.get(saveStart) == 0){
                              m.remove(saveStart);
                          }

                          i--;
                      }
                  }
              }

              return Math.max(ret, i-start);
    }
}
