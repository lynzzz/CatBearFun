int lengthOfLongestSubstringTwoDistinct(String s, int k){
 HashMap m = new HashMap<>();
   
   int i,j=0;
   int max = 0;;
   
   for( i=0; i<s.length();i++){
        while( j < s.length()){
		    char c = s.charAt(j);
		    if ( m.containsKey(c)){
			    m.put(c, m.get(c) + 1 );
			}else{
			    if ( m.size() == 2){
				    break;
				}
			    m.put(c, 1);
			}
			j++;
		}
   
        max = Math.max(max, j - i);
		
		char c = s.charAt(i);
		if ( m.containsKey(c)){
		    if ( m.get(c) > 1){
		        m.put(c, m.get(c) - 1);
		    }else{
		        m.remove(c);
		    }
		}
   }
   
   return max;
}
