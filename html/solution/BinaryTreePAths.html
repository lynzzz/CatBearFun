public class Solution {

public class Solution {
    public List binaryTreePaths(TreeNode root) {
        
		List ret = new ArrayList();
		helper(root, ret, "" );	
		return ret;
		
    }
    
    public void helper(TreeNode root, List results, String str){
	     
		 if ( root == null ){
		     return;
		 }
		 
		 
		 if ( root.left == null && root.right == null ){
		     results.add(str + String.valueOf(root.val) );
			 return;
		 }
		 
		 helper(root.left, results, str + String.valueOf(root.val) + "->" );
		 helper(root.right, results, str + String.valueOf(root.val) + "->");	 
	}
}
}
Time complexity : O(n)
Space complexity : Klg(n), where k is the number of path that meets the requirement
