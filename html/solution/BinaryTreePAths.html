class Solution {
      public List<String> binaryTreePaths(TreeNode root) {

          List<String> results = new ArrayList<String>();
          ArrayList<Integer> result = new ArrayList<>();
          if ( root == null) return results;


          result.add(root.val);
          helper(results, result, root);
          return results;
      }

      void helper(List<String> results, ArrayList<Integer> result, TreeNode root){
          if ( root.left == null && root.right == null){
              String str = "";
              for( int i=0; i<result.size(); i++){

                  if ( i == result.size() - 1){
                      str = str + String.valueOf(result.get(i));
                  }else{
                      str = str + String.valueOf(result.get(i)) + "->";
                  }
              }
              results.add(str);
              return;
          }

          if ( root.left != null){
              result.add(root.left.val);
              helper(results, result, root.left);
              result.remove(result.size() - 1);
          }

          if ( root.right != null){
              result.add(root.right.val);
              helper(results, result, root.right);
              result.remove(result.size() - 1);
          }

      }
  }
Time complexity : O(n)
Space complexity : Klg(n), where k is the number of path that meets the requirement
技巧 ： 1. 先把顶点放入小容器，好写一些， 2. String + 在recursion中不会带入其他递归层
