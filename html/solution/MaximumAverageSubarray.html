Solution one : sliding window 


public class Solution {
    /**
     * @param nums an array with positive and negative numbers
     * @param k an integer
     * @return the maximum average
     */
    public double maxAverage(int[] nums, int k) {
        // Write your code here
        
        double ret = getmaxAverage(nums, nums.length);
        
        for( int size = nums.length - 1; size>=k; size--){
            ret = Math.max(ret, getmaxAverage(nums, size));
        }
        
        return ret;
    }
    
    
    
    public double getmaxAverage(int[] nums, int k){
        
        int maxSum = 0;
        for( int startIndex = 0; startIndex <= nums.length - k; startIndex++){
            
            int sum = 0;
            for( int j=0; j<k; j++){
                sum = sum + nums[startIndex + j];
            }
            maxSum = Math.max(sum, maxSum);
        }
        
        return (double)maxSum/k;
    }
}
