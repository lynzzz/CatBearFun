        // Write your code here
        if ( nums == null || nums.length == 0) return -1;
        
        int start = 0;
        int end = nums.length - 1;
        
        while ( ( start + 1) < end ){
            
            int mid = start + (end - start)/2;
            
            if ( nums[mid] > target){
                end = mid;
            }else if ( nums[mid] < target){
