Solution I:: 遍历的写法

  class Solution {
      int count = 0;
      int saveSum = 0;
      public int pathSum(TreeNode root, int sum) {
          if ( root == null) return 0;

          saveSum = sum;

          helper(root);
          return count;
      }

      void helper(TreeNode root){     // inorder写法遍历每一个点， 把print换成DFS往里面套。
          if ( root == null) return;

          dfs(root, saveSum - root.val);

          helper(root.left);
          helper(root.right);
      }


      // 从顶点往下求和

      void dfs(TreeNode root, int sum){
          if ( sum == 0){
              count++;
          }

          if ( root.left != null){
              dfs(root.left, sum - root.left.val);
          }

          if ( root.right != null){
              dfs(root.right, sum- root.right.val);
          }
      }
  }
