class Solution {
        public int leastInterval(char[] tasks, int n) {
            int[] freq = new int[26];
            for( char c : tasks){
                freq[c -'A']++;
            }
            
            PriorityQueue pq = new PriorityQueue((a,b)->(b-a));
            
            for(int i : freq){
                if ( i != 0){
                    pq.add(i);
                }
            }
            
            int count = 0;
            
            while( !pq.isEmpty()){
                
                List ret = new ArrayList<>();
                int slot = n + 1;
    
                while( !pq.isEmpty() && slot > 0){
                    int top = pq.poll();
                    top--;
                    if ( top > 0){
                        ret.add(top);
                    }
                    
                    slot--;
                    count++;
                }
                
    
                for( int i : ret){
                    pq.add(i);
                }
                
                if ( pq.isEmpty()){
                    break;
                }
                
                count = count + slot;
    
            }
            
            return count;
        }
    }
