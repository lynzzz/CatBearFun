public class Solution {

    public ListNode reverse(ListNode head) {
        ListNode pre = null;
        ListNode current = head;

        
        while( current != null)
        {
            ListNode next = current.next;
            current.next = pre;
            pre = current;
            current = next;
        }

        // 结束以后，pre就是联表的头，cur是pre后面的那一个结点，在这里为null
        
        return pre;
    }
}
While 外部保存pre,cur指针， while 内部维持next指针
