    ArrayList mergeTwoSortedArray(ArrayList left, ArrayList right ){
        ArrayList ret = new ArrayList<>();
        
        int i=0;
        int j=0;
        int m = left.size();
        int n = right.size();
        while( i<m && j<n){
            if ( left.get(i) < right.get(j) ) {
                ret.add(left.get(i) );
                i++;
            }else{
                ret.add(right.get(j) );
                j++;
            }
        }
        
        while( i < m){
            ret.add(left.get(i) );
            i++;
        }
        
        while( j < n){
            
            ret.add( right.get(j) );
            j++;
        }
        return ret;
    }
