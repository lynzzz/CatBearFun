  class TrieNode{
      TrieNode[] children = new TrieNode[26];
      boolean isEnd = false;

      TrieNode getChild(char c){
          return children[c - 'a'];
      }

      void setChild(char c){
          children[c - 'a'] = new TrieNode();
      }

      boolean containChild(char c){
          return children[c-'a'] != null;
      }
  }

  class Trie {

      TrieNode root = new TrieNode();

      public Trie() {

      }

      public void insert(String word) {

          TrieNode cur = root;
          for( char c : word.toCharArray()){
              if ( !cur.containChild(c)){
                  cur.setChild(c);
              }
              cur = cur.getNode(c);
          }
          cur.isEnd = true;
      }

      public boolean search(String word) {

          TrieNode cur = root;

          for( char c : word.toCharArray()){
              if ( !cur.containChild(c)){
                  return false;
              }else{
                  cur = cur.getNode(c);
              }
          }
          return cur.isEnd;
      }

      public boolean startsWith(String prefix) {

          TrieNode cur = root;

          for( char c : prefix.toCharArray()){
              if ( !cur.contains(c)){
                  return false;
              }else{
                  cur = cur.getNode(c);
              }
          }

          return true;
      }
  }
    Time complexity : Insert O(m), where m is the length of the string, search O(m)
