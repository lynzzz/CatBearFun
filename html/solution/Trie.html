class Trie {

      TrieNode root = new TrieNode();


      public void insert(String word) {
          TrieNode node = root;
          for( char c : word.toCharArray()){
              if ( node.contains(c)){
                  node = node.get(c);                            // node.get(c) 直接走到自己的儿子了
              }else{
                  node.put(c, new TrieNode());
                  node = node.get(c);
              }
          }
          node.isEnd = true;
      }

      public TrieNode startsWithPrefix(String str){
          TrieNode node = root;
          for( char c : str.toCharArray()){
              if ( node.contains(c)){
                  node = node.get(c);
              }else{
                  return null;
              }
          }
          return node;
      }

      /** Returns if the word is in the trie. */
      public boolean search(String word) {
                  TrieNode n = startsWithPrefix(word);
                  return n!=null && n.isEnd;
      }

      /** Returns if there is any word in the trie that starts with the given prefix. */
      public boolean startsWith(String prefix) {
          TrieNode n = startsWithPrefix(prefix);
          return n!= null;
      }


  }


  class TrieNode{
      private TrieNode[] children = new TrieNode[26];
      public boolean isEnd;


      TrieNode get(char c){
          return children[c - 'a'];
      }

      void put(char c, TrieNode n){
               children[c - 'a'] = n;
      }

      boolean contains(char c){
          return children[c - 'a'] != null;
      }

  }
Time complexity : Insert O(m), where m is the length of the string, search O(m)
Space complexity : Insert worst case O(m)
