public class Solution {
    public int[] twoSum(int[] numbers, int target) {
        // write your code here
        HashMap m = new HashMap<>();
        
        int[] ret = new int[2];
        for( int i=0; i<numbers.length; i++){
            if ( m.containsKey(target - numbers[i]) ){
                ret[0] = m.get(target-numbers[i]) + 1;
                ret[1] = i+1;
                return ret;
            }
            
            m.put(numbers[i], i);
        }
        
        return ret;
    }
}
Time complexity : O(n)
Space complexity : O(n)

public class Solution {
    /*
     * @param numbers : An array of Integer
     * @param target : target = numbers[index1] + numbers[index2]
     * @return : [index1 + 1, index2 + 1] (index1 < index2)
     */
     
    class node{
        public int val;
        public int index;
        public node(int val, int index){
            this.val = val;
            this.index = index;
        }
    } 
    
    Comparator myComparator = new Comparator(){
      public int compare( node l, node r){
          if ( l == null) return 1;
          if ( r == null) return -1;
          return l.val - r.val;
      }
        
    };
     
    public int[] twoSum(int[] numbers, int target) {
        // write your code here
        int[] ret = new int[2];
        if ( numbers == null || numbers.length == 0) return ret;
        
        node[] arr = new node[numbers.length];
        for( int i=0; i< arr.length; i++){
            arr[i] = new node(numbers[i], i);
        }
        
        Arrays.sort(arr, myComparator);
        
        int i =0;
        int j = arr.length - 1;
        
        while( i < j){
            if ( arr[i].val + arr[j].val < target ){
                i++;
            }else if (arr[i].val + arr[j].val > target ){
                j--;
            }else{
                ret[0] = Math.min( arr[i].index+1, arr[j].index+1);
                ret[1] = Math.max( arr[i].index+1, arr[j].index+1);
                return ret;
            }
        }
        return ret;
    }
}
