public class Solution {
    public int maxSubArray(int[] nums) {
        // write your code
        if ( nums == null || nums.length == 0) return 0;
        
        int preFixSum = 0, preFixMinSum = 0, max = Integer.MIN_VALUE;
        
        for( int i=0; i<nums.length; i++){
            preFixSum = preFixSum + nums[i];
            max = Math.max(max, preFixSum - preFixMinSum);
            preFixMinSum = Math.min(preFixMinSum, preFixSum);
        }
        
        return max;
    }
}
