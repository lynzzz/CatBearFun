public int lengthOfLongestSubstring(String s) {

         HashMap<Character, Integer> m = new HashMap<>();

         int start = 0;
         int ret = 0;
         int i=0;

         for( ; i<s.length(); i++){
             if ( !m.containsKey(s.charAt(i)) ){
                 m.put(s.charAt(i), i);
             }else{
                 ret = Math.max(ret, i-start);

                 // 前面指针往前面走，一直走到后面指针可以往map里面加值为止。
                 while(m.containsKey(s.charAt(i))){
                     m.remove(s.charAt(start));
                     start++;
                 }
                 i--;
             }
         }

         return Math.max(ret, i-start);
     }
Two pointers moving forward 最长无重复子串
Time O(n)
