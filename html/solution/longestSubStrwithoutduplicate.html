public class Solution {
    /**
     * @param s: a string
     * @return: an integer 
     */
    public int lengthOfLongestSubstring(String s) {
        // write your code here
        int[] dict = new int[256];
        Arrays.fill(dict, -1);
        
        int length = 0;
        int max = 0;
        
        for( int i=0; i<s.length(); ){
            
            if ( dict[s.charAt(i)] == -1 ){
                dict[s.charAt(i)] = i;
                length++;
                i++;
            }else{
                max = Math.max(max, length);
                i = dict[s.charAt(i)] + 1;
                Arrays.fill(dict, -1);
                length = 0;
            }  
        }
        
        
        return Math.max(length, max);
    }
}
