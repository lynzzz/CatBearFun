    int getPosition(int[] table, int start, int end, int target){
        
        
        while((start + 1) < end){
            int mid = start + (end - start)/2;

            if ( table[mid] < target){
                start = mid;
            }else if ( table[mid] > target){
                end = mid;
            }else{
                return mid;
            }
        }

        if ( table[end] == target) return end;
        if ( table[start] == target) return start;

        if (table[start] > target) return start;

        return end;
    }
