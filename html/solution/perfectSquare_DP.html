public int numSquares(int n) {
        
        // def
        int[] dp = new int[n+1];                     // 开 n+1个数组的原因是让下标和输入的n一致，且后面的运算需要 
        
        // init
        dp[0] = 0;
        dp[1] = 1;                                   // 求最小值，初始化到最大 
        for( int i=2; i<n+1; i++){
            dp[i] = Integer.MAX_VALUE;
        }
                                                     //      i    =   j^2 + (i - j^2)     任何一个值都可以分解成一个完全平方数 + 一个非旺诠评反诉
        for( int i=2; i<n+1; i++){                   //    dp[i] = Math.min(dp[i]， 1 + dp[i-j^2])
            for( int j=1; j*j <= i ; j++){
                if ( ( dp[i-j*j] + 1) < dp[i] ){
                    dp[i] = dp[i-j*j] + 1;
                }
            }
        }
        
        return dp[n];
        
    }
Time complexity : n^(2)
