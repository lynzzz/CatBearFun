public class Solution {

    public int twoSumCloset(int[] nums, int target) {
        // Write your code here
        Arrays.sort(nums);
        int i=0, j=nums.length-1;
        int diff = Integer.MAX_VALUE;
        
        while( i < j){
            int sum = nums[i] + nums[j];
            
            if ( sum > target){
                if ( (  sum - target) < diff ) {
                    diff = sum - target;
                }
                j--;
            }
            else{
                if ( target - sum < diff){
                    diff = target - sum;
                }
                i++;
            }
        }
        return diff;
    }
}
