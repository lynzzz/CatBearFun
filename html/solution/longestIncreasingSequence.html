Solution I: DP n^2 

    class Solution {
        public int lengthOfLIS(int[] nums) {
            
            int[] dp = new int[nums.length];
            
            Arrays.fill(dp, 1);
            
            
            for(int i=1; i<nums.length; i++){
                for( int j=0; j<i; j++){
                    if ( nums[i] > nums[j]){
                        dp[i] = Math.max(dp[i], dp[j] + 1);
                    }
                }
            }
            
            int ret = Integer.MIN_VALUE;
            
            for( int i : dp){
                ret = Math.max(ret, i);
            }
            
            return ret;
            
        }
    }

    int getCurTailIndex(int[] table, int start, int end, int target){
        
        while(( start + 1) < end){
            
            int mid = start + (end -start)/2;
            
            if ( table[mid] < target){
                start = mid;
            }else if (table[mid] > target){
                end = mid;
            }else{
                return mid;
            }
        }
        
        if( table[end] == target) return end;
        
        if ( table[start] == target) return start;
        
        if ( table[start] > target) return start;
                
        return end;
    }
