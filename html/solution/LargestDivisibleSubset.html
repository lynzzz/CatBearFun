public List largestDivisibleSubset(int[] nums) {
        // write your code here
        if ( nums == null || nums.length == 0){
            return new ArrayList();
        }
        
        // check corner case
        Arrays.sort(nums);
        
        // Def: 
        int n = nums.length;
        
        int[] count = new int[n];                // Largest divisible subset count at nums[i]
        int[] pre = new int[n];                  // the next index of the next qualified integer
        
        
        // init:
        for( int i = 0; i<n ; i++){
            count[i] = 1;
            pre[i] = -1;
        }
        
        
        // function
        for( int i=1; i<n; i++){
            for ( int j=i-1; j>=0; j--){
                if ( nums[i]%nums[j] == 0 && ( (count[j] + 1) > count[i] ) ){
                    count[i] = count[j] + 1;
                    pre[i] = j;
                }
            }
        }
        
        
        // ret
        int maxIndex = -1;
        int max = Integer.MIN_VALUE;
        for( int i=0; i max){
                max = count[i];
                maxIndex = i;
            }
        }
        
        ArrayList ret = new ArrayList<>();
        
        while( maxIndex != -1){
            ret.add(nums[maxIndex]);
            maxIndex = pre[maxIndex];
        }
        
        return ret;
        
    }
