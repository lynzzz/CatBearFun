public int minPathSum(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        
        // Definition : dp(i,j ) means the minimum value from grid[0][0] to grid[i][j]
        int[][] dp = new int[m][n];
        
        // initial
        dp[0][0] = grid[0][0];

        for( int i=1; i<m; i++){
             dp[i][0] = dp[i-1][0] + grid[i][0];
        }
        
        for( int i=1; i<n; i++){
            dp[0][i] = dp[0][i-1] + grid[0][i];
        }
        
        
        // transfer
        for( int i=1; i<m; i++){
            for( int j=1; j<n; j++){
                dp[i][j] = Math.min(dp[i-1][j], dp[i][j-1]) + grid[i][j];
            }
        }
        
        // result
        return dp[m-1][n-1];
    }
