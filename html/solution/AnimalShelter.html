Solution ii : using 一条链， 题目转变为一个自定义的queue上删除任意点。 


public class AnimalShelter {

   class Node{
      String name;
      int type;
      Node next;

      Node(String name, int type){
          this.name = name;
          this.type = type;
      }
   }
 
    
   Node front;
   Node tail;

    public AnimalShelter() {
        // do initialize if necessary
    }

    /**
     * @param name a string
     * @param type an integer, 1 if Animal is dog or 0
     * @return void
     */
    void enqueue(String name, int type) {
        // Write your code here
        Node n = new Node(name, type);
        if ( front == null){
            front = n;
            tail = n;
        }else{
            tail.next = n;
            tail = n;
        }
    }

    public String dequeueAny() {
        // Write your code here
        String ret = front.name;
        front = front.next;
        return ret;
    }

    public String dequeueDog() {
        // Write your code here
        Node dummy = new Node("", 0);
        dummy.next = front;
        Node save = dummy;
        
        
        while( dummy.next != null ){
            if ( dummy.next.type == 1){
                
                if ( dummy.next.next == null){
                    tail = dummy;
                }
                
                String ret = dummy.next.name;
                dummy.next = dummy.next.next;
                front = save.next;
                return ret;
            }else{
                dummy = dummy.next;
            }
        }
        
        return "";
    }

    public String dequeueCat() {
        // Write your code here
        Node dummy = new Node("", 1);
        dummy.next = front;
        Node save = dummy;
        
        while( dummy.next != null){
            if ( dummy.next.type == 0){
                
                if ( dummy.next.next == null){
                    tail = dummy;
                }
                
                String ret = dummy.next.name;
                dummy.next = dummy.next.next;
                front = save.next;
                return ret;
            }else{
                dummy = dummy.next;
            }
        }
        
        return "";
    }
}
