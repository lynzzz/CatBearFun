public class Solution {
    public int twoSum6(int[] nums, int target) {
        // Write your code here
        
        Arrays.sort(nums);
        
        int i =0;
        int j = nums.length - 1;
        int count = 0;
        
        while ( i < j){
            
            int sum = nums[i] + nums[j];
            
            
            if ( sum > target){
                j--;
            }else if ( sum < target){
                i++;
            }else{
                
                count++;
                i++;
                j--;
                
                while ( i < j && nums[i] == nums[i-1] ){
                    i++;
                }
            
                while(  i < j && nums[j] == nums[j+1] ){
                    j--;
                }
            

            }
        }
        return count;
    }
}
