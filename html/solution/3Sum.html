public class Solution {
    public ArrayList> threeSum(int[] numbers) {
        // write your code here
        ArrayList> ret = new ArrayList<>();
        if ( numbers == null || numbers.length == 0) return ret;
        
        
        Arrays.sort(numbers);
        for( int i=0; i< numbers.length - 2; ){   [4] 自增符号去掉
            int j=i+1;
            int k=numbers.length - 1;
            while( j < k){
                if ( ( numbers[i] + numbers[j] + numbers[k]) == 0 ){
                    ArrayList temp = new ArrayList<>();
                    temp.add(numbers[i]);
                    temp.add(numbers[j]);
                    temp.add(numbers[k]);
                    ret.add(temp);
                    
					// [1]j去重
                    j++;
                    while( j < k && numbers[j] == numbers[j-1]){
                        j++;
                    }
                    
					// [2] k去重
                    k--;
                    while( k > j && numbers[k] == numbers[k+1] ){
                        k--;
                    }
                }
                else if ( ( numbers[i] + numbers[j] + numbers[k] ) > 0  ){
                    k--;
                }else{
                    j++;
                }
            }
            
			// [3] i去重
            i++;
            while ( i< numbers.length - 2 && numbers[i] == numbers[i-1]){
                i++;
            }
        }
        
        return ret;
    }
}
