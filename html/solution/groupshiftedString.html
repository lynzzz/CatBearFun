  class Solution {
      public List> groupStrings(String[] strings) {
          HashMap<Integer, List<String>> m = new HashMap<>();

          for( int i=0; i<strings.length; i++){
              String str = strings[i];
              int size = str.length();

              if ( !m.containsKey(size)){
                  List l = new ArrayList<>();
                  l.add(str);
                  m.put(size, l);
              }else{
                  m.get(size).add(str);
              }
          }

          List<List<String>> results = new ArrayList<>();

          for( int key : m.keySet()){
              List strs = m.get(key);

              int start = 0;
              while( start < strs.size() ){

                   List result = new ArrayList<>();


                   String firstString = strs.get(start);
                   result.add(firstString);

                   int count  = 0;
                   for( int i=start + 1; i<strs.size();i++){
                       if ( areTheSameGroup( firstString, strs.get(i) ) ){
                           result.add(strs.get(i));
                       }else{
                           count++;
                           if ( count == 1){
                               start = i;
                           }
                       }
                   }
                   results.add(result);
              }
          }

          return results;
      }


      boolean areTheSameGroup(String str1, String str2){
          if ( str1.length() == 1 && str2.length() == 1) return true;

          int diff = str1.charAt(0) - str2.charAt(0);

          for( int i=1; i<str1.length() ; i++){
              int delta = str1.charAt(i) - str2.charAt(i);
              if ( diff != delta){
                  return false;
              }
          }

          return true;
      }
  }









Time complexity :  O(n^2*l), where l is the average length of each string    

Space complexity : O(n)    





Solution II :看了Leetcode答案， 每个单词变换为以'a'开头的词，做key,放入hashmap， key一样就是一组的

class Solution {
    public List<List<String>> groupStrings(String[] strings) {
        List<List<String>> results = new ArrayList<>();
        HashMap<String, List<String>> m = new HashMap<>();

        for( String str : strings){

            int offset = str.charAt(0) - 'a';
            StringBuilder sb = new StringBuilder();
            for( char c : str.toCharArray()){
                char transformC = (char)(c - offset);
                if ( transformC < 'a'){
                    transformC = (char)(transformC + 26);
                }
                sb = sb.append(transformC);
            }

            String key = sb.toString();

            if ( !m.containsKey(key)){
                List l = new ArrayList<>();
                l.add(str);
                m.put(key, l);
            }else{
                m.get(key).add(str);
            }
        }


        for( String key : m.keySet()){
            List l = m.get(key);
            results.add(l);
        }

        return results;
    }
}
Time complexity : O(n*l), where l is the average length of each string
Space complexity : O(n)
