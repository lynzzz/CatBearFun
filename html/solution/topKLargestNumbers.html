class Solution {
    public int[] topk(int[] nums, int k) {
        // Write your code here
        PriorityQueue pq = new PriorityQueue();
        for(int i=0; i<k; i++){
            pq.add(nums[i]);
        }
        
        for( int i=k; i pq.peek() ){
                pq.remove();
                pq.add(nums[i]);
            }
        }
        
        int[] ret = new int[k];
        for( int i=k-1; i>=0; i--){
            ret[i] = pq.poll();
        }
        
        return ret;
    }
};
