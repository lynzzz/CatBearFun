class Solution {
    public String nextClosestTime(String time) {
        
        if ( time.equals("00:00")){
            return "00:00";
        }
        
        String[] hourAndMin = time.split(":");
        String str = hourAndMin[0] + hourAndMin[1];
        
        HashSet<Character> set = new HashSet<>();        
        for( char c : str.toCharArray()){
            set.add(c);
        }
                
        int number = Integer.valueOf(str);

        
        
        String ans = "";
        while( true  ){
            
            number++;
            if ( number == 2400){
                number = 0;
            }
            
            ans = String.valueOf(number);
            if ( ans.length() != 4){
                int diff = 4 - ans.length();
                for( int i=0; i<diff; i++){
                    ans = '0' + ans;
                }
            }
            
            if ( isTimeStringValid( ans ) && canStringBeReused(ans, set) ){
                break;
            }
        }
        

        
        
        StringBuilder sb = new StringBuilder(ans);
        sb.insert(2, ':');
        
        return sb.toString();
    }
        
    
    
    
    boolean isTimeStringValid(String number){
            
            
            char[] str = number.toCharArray();
            
            if ( str[0] < '2'){
                return str[1] <='9' && str[2]<='5' && str[3]<='9';
            }
            
            return str[1] <='3' && str[2]<='5' && str[3]<='9';
        }
        
    boolean canStringBeReused( String number, HashSet set){
           
            for( char c : number.toCharArray()){
                if ( !set.contains(c)){
                    return false;
                }
            }
            
            return true;
    }
}

Solution II : 转换成分钟计算，更简单
算法 ：把当前时间转换成分钟，自增，然后mod(24*60), 保证了值一直在有效范围内变化。 然后反算四位数。
hour = cur/60, minute = cur%60  结果的第一位为ret/10, 结果的第二位为ret%10

class Solution {
    public String nextClosestTime(String time) {
        
        String[] strs = time.split(":");
        int curMinute = Integer.valueOf(strs[0])*60 + Integer.valueOf(strs[1]);
        
        HashSet<Integer> set = new HashSet<>();
        for( String str : strs){
            for ( char c : str.toCharArray()){
                set.add(c - '0');
            }
        }
        
        StringBuilder sb = null;
        while( true){
            curMinute = (curMinute + 1)%(24 * 60);
            
            int[] numbers = new int[] { curMinute/60/10, curMinute/60%10, curMinute%60/10, curMinute%60%10 };

            int i = 0;
            for( ; i<numbers.length; i++){
                if ( !set.contains(numbers[i])){
                    break;
                }
            }
            
            if ( i != numbers.length){
                continue;
            }
            
            sb = new StringBuilder();
            for( int n : numbers){
                sb.append(n);
            }
            sb.insert(2, ':');
            return sb.toString();

        }
        

        
        
    }
}
