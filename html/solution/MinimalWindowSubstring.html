public String minWindow(String source, String target) {
        // write your code
        
        // build reference dict
        int[] dict = new int[256];
        for( int i=0; i<target.length(); i++ ){
            char c = target.charAt(i);
            dict[c]++;
        }
        
        
        int[] srcDict = new int[256];

        int i = 0;
        int start = 0;
        int count = 0;
        String ret = source;
        boolean found = false;
        

        for( ; i dict[source.charAt(start)] ){
                    srcDict[source.charAt(start)]--;
                    start++;
                }
                
                
                // update string
                if ( ( i - start + 1) < ret.length() ){
                    ret = source.substring(start, i+1);
                }

                //破坏这个平衡，头尾向前走一步 
                srcDict[source.charAt(start)]--;
                start++;
                count--;
            }
        }
        
        return found ? ret : "";
    }
