class Solution {
    
    int area = 0;
    int max = 0;
    
    public int maxAreaOfIsland(int[][] grid) {
        
        
        // check grid
        
        for( int i=0; i<grid.length; i++){
            for( int j=0; j<grid[0].length; j++){
                if ( grid[i][j] == 1){
                    dfs(grid, i, j);
                    max = Math.max(max, area);
                    area = 0;
                }
            }
        }
        
        return max;
    }
    
    void dfs(int[][] grid, int r, int c){
        
        int row = grid.length;
        int col = grid[0].length;
        if ( r < 0 || r==row || c < 0 || c==col){
            return;
        }
        
        if ( grid[r][c] == 0){
            return;
        }
        
        grid[r][c] = 0;            // 去重
        
        area++;
        dfs(grid, r-1, c);
        dfs(grid, r+1, c);
        dfs(grid, r, c+1);
        dfs(grid, r, c-1);
    }
}
