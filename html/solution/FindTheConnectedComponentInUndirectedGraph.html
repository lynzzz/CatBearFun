public class Solution {
    /**
     * @param nodes a array of Undirected graph node
     * @return a connected set of a Undirected graph
     */
    public List> connectedSet(ArrayList nodes) {
        // Write your code here
        HashSet set = new HashSet<>();
        Queue q = new LinkedList<>();
        
        List> results = new ArrayList<>();
        
        for( UndirectedGraphNode n : nodes ){
            
            if ( !set.contains(n.label) ){
                
                List result = new ArrayList<>();
                set.add(n.label);
                q.add(n);
            
                while( !q.isEmpty() ){
                
                    UndirectedGraphNode tempNode = q.poll();
                    result.add(tempNode.label);
                    for( UndirectedGraphNode tempNodeNeighbor : tempNode.neighbors){
                        if ( !set.contains(tempNodeNeighbor.label) ){
                            set.add(tempNodeNeighbor.label);
                            q.add(tempNodeNeighbor);
                        }
                    }
                }
                Collections.sort(result);
                results.add(new ArrayList(result));
                
                while ( !q.isEmpty() ){
                    q.remove();
                }
            }
        }
        
        return results;
    }
}
