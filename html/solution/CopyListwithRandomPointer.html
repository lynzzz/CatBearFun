public class Solution {
    public RandomListNode copyRandomList(RandomListNode head) {
        HashMap m = new HashMap<>();
        
        RandomListNode SaveHead = head;
        
        //Copy vertices
        while(head != null){
            m.put(head, new RandomListNode(head.label));
            head = head.next;
        }
        

        head = SaveHead;
        while(head != null){
            
            // connect next edge
            m.get(head).next = m.get(head.next);
            
            // connect random edge
            if ( head.random != null){
                m.get(head).random = m.get(head.random);
            }
            head = head.next;
        }
        
        return m.get(SaveHead);
    }
    
}
