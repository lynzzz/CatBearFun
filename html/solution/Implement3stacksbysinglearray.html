public class ThreeStacks {

    private int top0 = -1;
    private int top1 = -1;
    private int top2 = -1;
    
    private int start0;
    private int start1;
    private int start2;
    
    private int stackSize;
    private int arr[];

    public ThreeStacks(int size) {
        // do intialization if necessary
        arr = new int[3*size];
        stackSize = size;
        start0 = 0;
        start1 = size;
        start2 = 2*size;
    }

    public void push(int stackNum, int value) {
        // Write your code here
        // Push value into stackNum stack
        if ( stackNum == 0){
            if ( top0 == -1){
              top0 = 0;  
            } else{
                top0++;
            }
            arr[top0] = value;
        }else if ( stackNum == 1){
            if ( top1 == -1){
              top1 = stackSize;  
            } else{
                top1++;
            }
            arr[top1] = value;
        }else{
            if ( top2 == -1){
              top2 = 2*stackSize;  
            } else{
                top2++;
            }
            arr[top2] = value;
        }
    }
    
    public int pop(int stackNum) {
        // Write your code here
        // Pop and return the top element from stackNum stack
        if ( stackNum == 0){
            int ret = arr[top0];
            if ( top0 == start0){
                top0 = -1;
            }else{
                top0--;
            }
            return ret;
        }else if ( stackNum == 1){
            int ret = arr[top1];
            if ( top1 == start1){
                top1 = -1;
            }else{
                top1--;
            }
            return ret;
        }
            int ret = arr[top2];
            if ( top2 == start2){
                top2 = -1;
            }else{
                top2--;
            }
            return ret;
    }

    public int peek(int stackNum) {
        // Write your code here
        // Return the top element
        if ( stackNum == 0){
            return arr[top0];
        }else if ( stackNum == 1){
            return arr[top1];
        }
        return arr[top2];
    }

    public boolean isEmpty(int stackNum) {
        // Write your code here
        if ( stackNum == 0){
            return top0 == -1;
        }else if ( stackNum == 1){
            return top1 == -1;
        }
        return top2 == -1;
    }
}
