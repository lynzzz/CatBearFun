    /**

     * @return an integer
     */
     
     
    public int numIslands(boolean[][] grid) {
        // Write your code here
        if ( grid == null || grid.length == 0) return 0;
        
        int count = 0;
        
    
        for( int i=0; i<grid.length; i++){
            for( int j = 0; j q = new LinkedList<>();

        int code = i * col + j;
        
        
        q.add(code);

        
        while ( !q.isEmpty() ){
            
            code  = q.poll();
            i = code/col;
            j = code%col;
            grid[i][j] = false;


            // up 
            if (  (i-1)>=0  && grid[i-1][j]  ){
                q.add(   (i-1) * col + j);
            }
            
            // down
            if ( (i + 1) < grid.length && grid[i+1][j]  ){
                q.add(   (i+1) * col + j );

            }
            
                        
            // left
            if ( (j-1) >= 0 && grid[i][j-1]){
                q.add(i*col + j-1);

            }
            
            // right
            if ( (j + 1) < col && grid[i][j+1] ){
                q.add( i*col + j+1);
            }
        }
    }
}
