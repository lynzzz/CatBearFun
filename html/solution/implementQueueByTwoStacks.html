public class Queue {
    private Stack stack1;
    private Stack stack2;

    public Queue() {
       // do initialization if necessary
       stack1 = new Stack();
       stack2 = new Stack();
    }
    
    public void push(int element) {
        // write your code here
        stack1.push(element);
    }

    public int pop() {
        // write your code here
        while( stack1.size() != 1){
            stack2.push(stack1.pop());
        }
        
        int ret = stack1.pop();
        
        while( stack2.size()!= 0){
            stack1.push(stack2.pop());
        }
        return ret;
    }

    public int top() {
        // write your code here
        while( stack1.size() != 1){
            stack2.push(stack1.pop());
        }
        
        int ret = stack1.peek();
        while( stack2.size() != 0 ){
            stack1.push(stack2.pop());
        }
        
        return ret;
    }
}
