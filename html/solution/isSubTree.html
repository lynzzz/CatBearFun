class Solution {
    public boolean isSubtree(TreeNode s, TreeNode t) {                  // 点对树模板
        
         if ( s == null) return false;
         if ( areTheSameTree(s, t)) return true;
        
         return isSubtree(s.left, t) || isSubtree(s.right, t);
    }
    
    
    boolean areTheSameTree(TreeNode t1, TreeNode t2){                  //树对树模板
        
        if ( t1 == null && t2 == null) return true;
        if ( t1 == null || t2 == null ) return false;
        if ( t1.val != t2.val ) return false;
        
        return areTheSameTree(t1.left, t2.left) && areTheSameTree(t1.right, t2.right);
    }
    
    
    
}
