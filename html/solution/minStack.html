public class MinStack {
    
    class Node{
        int val;
        Node next;
        public Node( int val){
            this.val = val;
        }
    }
    
    private Node mainTop;
    private Node minTop;
    
    public MinStack() {
        // do initialize if necessary
    }

    public void push(int number) {
        // write your code here
        Node mainNode = new Node(number);
        Node minNode = new Node(number);
        
        if ( mainTop == null ){
            mainTop = mainNode;
            minTop = minNode;
        }else{
            mainNode.next = mainTop;
            mainTop = mainNode;
            
            if ( minNode.val <= minTop.val){
                minNode.next = minTop;
                minTop = minNode;
            }
        }
        
    }

    public int pop() {
        // write your code here
        int ret = mainTop.val;
        mainTop = mainTop.next;
        
        if ( ret == minTop.val){
            minTop = minTop.next;
        }
        return ret;
    }

    public int min() {
        // write your code here
        return minTop.val;
    }
}
Time complexity : O(1) for all
Space complexity : one more stack
